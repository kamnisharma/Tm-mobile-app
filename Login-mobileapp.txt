<script>
  // ---------- STORAGE KEYS ----------
  const STORAGE_USERS_KEY = "flowauth_registered_users";
  const SESSION_USER_KEY = "flowauth_current_user";

  let registeredUsers = [];

  function loadUsersFromStorage() {
    const stored = localStorage.getItem(STORAGE_USERS_KEY);
    if (stored) {
      try {
        registeredUsers = JSON.parse(stored);
        if (!Array.isArray(registeredUsers)) registeredUsers = [];
      } catch(e) { registeredUsers = []; }
    }
    if (!registeredUsers.length) {
      registeredUsers = [{ username: "demo", password: "demo123" }];
      saveUsersToStorage();
    }
  }

  function saveUsersToStorage() {
    localStorage.setItem(STORAGE_USERS_KEY, JSON.stringify(registeredUsers));
  }

  function addUser(username, password) {
    const exists = registeredUsers.some(u => u.username === username);
    if (exists) return false;
    registeredUsers.push({ username, password });
    saveUsersToStorage();
    return true;
  }

  function validateUser(username, password) {
    const user = registeredUsers.find(u => u.username === username);
    return user && user.password === password;
  }

  // session mgmt
  function isLoggedIn() {
    return sessionStorage.getItem(SESSION_USER_KEY) !== null;
  }

  function getCurrentSessionUser() {
    const raw = sessionStorage.getItem(SESSION_USER_KEY);
    if (raw) {
      try {
        return JSON.parse(raw);
      } catch(e) { return null; }
    }
    return null;
  }

  function setLoggedInUser(username) {
    sessionStorage.setItem(SESSION_USER_KEY, JSON.stringify({ username }));
  }

  function logoutUser() {
    sessionStorage.removeItem(SESSION_USER_KEY);
  }

  // DOM elements
  const authWrapper = document.getElementById('authWrapper');
  const landingView = document.getElementById('landingView');
  const loginView = document.getElementById('loginView');
  const signupView = document.getElementById('signupView');
  const loginForm = document.getElementById('loginForm');
  const signupForm = document.getElementById('signupForm');
  const loginMessageDiv = document.getElementById('loginMessage');
  const signupMessageDiv = document.getElementById('signupMessage');

  function showLoginMessage(text, isError = true) {
    if (loginMessageDiv) {
      loginMessageDiv.innerHTML = text;
      loginMessageDiv.className = `message-area ${isError ? 'error-msg' : 'success-msg'}`;
    }
  }

  function showSignupMessage(text, isError = true) {
    if (signupMessageDiv) {
      signupMessageDiv.innerHTML = text;
      signupMessageDiv.className = `message-area ${isError ? 'error-msg' : 'success-msg'}`;
    }
  }

  // update UI based on login state and hash
  function updateUIBasedOnAuth() {
    const loggedIn = isLoggedIn();
    if (loggedIn) {
      // show beautiful landing, hide auth card
      authWrapper.style.display = 'none';
      landingView.style.display = 'block';
      
      const sessionUser = getCurrentSessionUser();
      const username = sessionUser ? sessionUser.username : 'User';
      // update landing text fields
      const landingUsernameSpan = document.getElementById('landingUsername');
      const heroUserNameSpan = document.getElementById('heroUserName');
      const userAvatarDiv = document.getElementById('userAvatar');
      if (landingUsernameSpan) landingUsernameSpan.innerText = username;
      if (heroUserNameSpan) heroUserNameSpan.innerText = username;
      if (userAvatarDiv) userAvatarDiv.innerText = username.charAt(0).toUpperCase();
    } else {
      authWrapper.style.display = 'flex';
      landingView.style.display = 'none';
      // decide login or signup based on hash
      const hash = window.location.hash;
      if (hash === '#signup') {
        loginView.style.display = 'none';
        signupView.style.display = 'block';
      } else {
        loginView.style.display = 'block';
        signupView.style.display = 'none';
        if (window.location.hash !== '#login') window.location.hash = '#login';
      }
      clearAuthMessages();
    }
  }

  function clearAuthMessages() {
    if (loginMessageDiv) loginMessageDiv.innerHTML = '';
    if (signupMessageDiv) signupMessageDiv.innerHTML = '';
    if (loginMessageDiv) loginMessageDiv.className = 'message-area';
    if (signupMessageDiv) signupMessageDiv.className = 'message-area';
  }

  // Login handler
  function handleLogin(e) {
    e.preventDefault();
    if (isLoggedIn()) {
      updateUIBasedOnAuth();
      return;
    }
    const username = document.getElementById('loginUsername').value.trim();
    const password = document.getElementById('loginPassword').value;
    if (!username || !password) {
      showLoginMessage('❌ Please fill in both fields', true);
      return;
    }
    if (validateUser(username, password)) {
      setLoggedInUser(username);
      document.getElementById('loginUsername').value = '';
      document.getElementById('loginPassword').value = '';
      showLoginMessage('✅ Login successful! Redirecting...', false);
      updateUIBasedOnAuth();
    } else {
      showLoginMessage('❌ Invalid credentials. Try demo/demo123', true);
    }
  }

  // Signup handler
  function handleSignup(e) {
    e.preventDefault();
    if (isLoggedIn()) {
      updateUIBasedOnAuth();
      return;
    }
    const username = document.getElementById('signupUsername').value.trim();
    const password = document.getElementById('signupPassword').value;
    const confirm = document.getElementById('signupConfirm').value;
    if (!username || !password || !confirm) {
      showSignupMessage('⚠️ All fields required', true);
      return;
    }
    if (password.length < 4) {
      showSignupMessage('🔒 Password must be at least 4 characters', true);
      return;
    }
    if (password !== confirm) {
      showSignupMessage('❌ Passwords do not match', true);
      return;
    }
    if (registeredUsers.some(u => u.username === username)) {
      showSignupMessage(`⚠️ Username "${username}" already exists`, true);
      return;
    }
    const added = addUser(username, password);
    if (added) {
      showSignupMessage('🎉 Account created! Redirect to login...', false);
      document.getElementById('signupUsername').value = '';
      document.getElementById('signupPassword').value = '';
      document.getElementById('signupConfirm').value = '';
      setTimeout(() => {
        window.location.hash = '#login';
        updateUIBasedOnAuth();
      }, 1200);
    } else {
      showSignupMessage('⚠️ Signup error, try again', true);
    }
  }

  // logout from landing
  function logoutFromLanding() {
    logoutUser();
    window.location.hash = '#login';
    updateUIBasedOnAuth();
  }

  // handle beautiful landing buttons with interactive toasts / alerts (dummy actions)
  function attachLandingButtons() {
    const buttons = document.querySelectorAll('.action-btn');
    buttons.forEach(btn => {
      btn.removeEventListener('click', handleButtonAction);
      btn.addEventListener('click', handleButtonAction);
    });
    const logoutLanding = document.getElementById('logoutLandingBtn');
    if (logoutLanding) {
      logoutLanding.removeEventListener('click', logoutFromLanding);
      logoutLanding.addEventListener('click', logoutFromLanding);
    }
  }

  function handleButtonAction(e) {
    const btn = e.currentTarget;
    const action = btn.getAttribute('data-action') || 'action';
    // create a small floating notification effect
    const fakeToast = document.createElement('div');
    fakeToast.innerText = `✨ ${action.charAt(0).toUpperCase() + action.slice(1)} clicked — demo interaction`;
    fakeToast.style.position = 'fixed';
    fakeToast.style.bottom = '20px';
    fakeToast.style.left = '50%';
    fakeToast.style.transform = 'translateX(-50%)';
    fakeToast.style.backgroundColor = '#1e293b';
    fakeToast.style.color = 'white';
    fakeToast.style.padding = '10px 20px';

  // hash change
  function onHashChange() {
    if (!isLoggedIn()) {
      const hash = window.location.hash;
      if (hash === '#signup') {
        loginView.style.display = 'none';
        signupView.style.display = 'block';
      } else {
        loginView.style.display = 'block';
        signupView.style.display = 'none';
      }
      clearAuthMessages();
    } else {
      updateUIBasedOnAuth();
    }
  }

  function init() {
    loadUsersFromStorage();
    loginForm.addEventListener('submit', handleLogin);
    signupForm.addEventListener('submit', handleSignup);
    const signupLink = document.getElementById('signupLinkNav');
    const loginLink = document.getElementById('loginLinkNav');
    if (signupLink) signupLink.addEventListener('click', (e) => { e.preventDefault(); window.location.hash = '#signup'; onHashChange(); });
    if (loginLink) loginLink.addEventListener('click', (e) => { e.preventDefault(); window.location.hash = '#login'; onHashChange(); });
    window.addEventListener('hashchange', onHashChange);
    if (!window.location.hash || (window.location.hash !== '#login' && window.location.hash !== '#signup')) {
      window.location.hash = '#login';
    } else {
      onHashChange();
    }
    updateUIBasedOnAuth();
    // after each render we reattach landing buttons (observer not needed, we call after updates)
    const observer = new MutationObserver(() => {
      if (landingView.style.display === 'block') attachLandingButtons();
    });
    observer.observe(document.getElementById('appRoot'), { attributes: true, subtree: false, childList: true });
    attachLandingButtons();
  }

  document.addEventListener('DOMContentLoaded', init);
